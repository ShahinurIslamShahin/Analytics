SELECT /*+parallel(p,16)*/
        TO_NUMBER(SUBSTR(TIMEENTERACTIVE,1,4)) YEAR,TO_NUMBER(SUBSTR(TIMEENTERACTIVE,5,2)) MONTH,
        CONCAT('880',SUBSTR(MSISDN,1,5)) SERIES, COUNT (UNIQUE MSISDN) TOTAL_ACTIVE_TILL_NOW
   FROM DWH_USER.L1_FUO_OCS_SUBSCRIBER_ALL@DWH05TODWH01 P
         INNER JOIN DWH_USER.PRODUCT_DIM ON P.SUBCOSID = PRODUCT_ID
   WHERE ACTIVEFLAG = 1 AND USERSTATE = 2
GROUP BY TO_NUMBER(SUBSTR(TIMEENTERACTIVE,1,4)),TO_NUMBER(SUBSTR(TIMEENTERACTIVE,5,2)),SUBSTR(MSISDN,1,5)
ORDER BY TO_NUMBER(SUBSTR(TIMEENTERACTIVE,1,4)),TO_NUMBER(SUBSTR(TIMEENTERACTIVE,5,2)),SUBSTR(MSISDN,1,5)
